Implementation

Add an @theme block to src/styles/global.css after @import "tailwindcss". In Tailwind v4, variables in @theme are registered as utility classes automatically: --color-accent becomes text-accent, bg-accent, border-accent; --font-heading becomes font-heading; --radius-lg becomes rounded-lg; and so on.

The --role-* semantic aliases and --neutral-* raw palette stay in :root — they're internal references, not utility classes. 

/* src/styles/global.css */
@import "tailwindcss";

@plugin "daisyui" { themes: lofi --default; }

/* ──────────────────────────────────────────────────────────
   Design tokens. Tailwind v4 exposes @theme vars as utilities.
   Change a value here; it propagates everywhere.
   ────────────────────────────────────────────────────────── */
@theme {
  /* Fonts → font-body, font-heading, font-mono */
  --font-body:    'DM Sans', sans-serif;
  --font-heading: 'Newsreader', serif;
  --font-mono:    'Share Tech Mono', monospace;

  /* Type scale → text-xs … text-5xl (overrides Tailwind defaults) */
  --text-xs:   0.64rem;
  --text-sm:   0.8rem;
  --text-base: 1rem;
  --text-lg:   1.25rem;
  --text-xl:   1.563rem;
  --text-2xl:  1.953rem;
  --text-3xl:  2.441rem;
  --text-4xl:  3.052rem;
  --text-5xl:  3.815rem;

  /* Colors → text-text, bg-surface, border-border, text-accent … */
  --color-text:          oklch(10%  0.005 250);
  --color-text-muted:    oklch(48%  0.010 250);
  --color-text-subtle:   oklch(58%  0.010 250);
  --color-text-disabled: oklch(68%  0.010 250);
  --color-bg:            #ffffff;
  --color-surface:       oklch(98%  0.002 250);
  --color-border:        oklch(88%  0.006 250);
  --color-border-strong: oklch(78%  0.008 250);
  --color-accent:        oklch(50%  0.15  252);
  --color-accent-hover:  oklch(38%  0.13  252);

  /* Radius → rounded-sm, rounded-md, rounded-lg, rounded-xl */
  --radius-sm:   0.25rem;
  --radius-md:   0.5rem;
  --radius-lg:   0.75rem;
  --radius-xl:   1.25rem;
  --radius-full: 9999px;

  /* Shadows → shadow-sm, shadow-md, shadow-lg */
  --shadow-sm: 0 1px 2px 0 oklch(0% 0 0 / 0.04);
  --shadow-md: 0 4px 12px -2px oklch(0% 0 0 / 0.07);
  --shadow-lg: 0 8px 24px -4px oklch(0% 0 0 / 0.12);

  /* Motion → duration-fast, duration-base, duration-slow */
  --duration-fast:  100ms;
  --duration-base:  200ms;
  --duration-slow:  400ms;
  --ease-out:    cubic-bezier(0.22, 1, 0.36, 1);
  --ease-in-out: cubic-bezier(0.40, 0, 0.20, 1);
  --ease-spring: cubic-bezier(0.34, 1.56, 0.64, 1);
}

/* Semantic aliases — internal references, not utility classes */
:root {
  /* Type roles — change here to remap a role site-wide */
  --role-label:   var(--text-sm);
  --role-body:    var(--text-base);
  --role-lead:    var(--text-lg);
  --role-h3:      var(--text-xl);
  --role-h2:      var(--text-2xl);
  --role-h1:      var(--text-3xl);
  --role-display: var(--text-4xl);

  /* Semantic spacing */
  --space-element: 1rem;    /* was: gap-4 hardcoded */
  --space-block:   2rem;    /* was: gap-8 hardcoded */
  --space-section: 5rem;    /* was: gap-20 hardcoded */

  /* Z-index ladder */
  --z-raised:  10;
  --z-overlay: 20;
  --z-modal:   30;
  --z-toast:   40;
  --z-tooltip: 50;
}
